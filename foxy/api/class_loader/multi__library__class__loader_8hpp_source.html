89  template<class Base>

91  {
92  CONSOLE_BRIDGE_logDebug(
93  "class_loader::MultiLibraryClassLoader: "
94  "Attempting to create instance of class type %s.",

99  "MultiLibraryClassLoader: Could not create object of class type " +

101  " as no factory exists for it. Make sure that the library exists and " +

103  }

308  template<typename Base>
309  ClassLoader * getClassLoaderForClass(const std::string & class_name)
310  {
311  ClassLoaderVector loaders = getAllAvailableClassLoaders();
312  for (ClassLoaderVector::iterator i = loaders.begin(); i != loaders.end(); ++i) {
313  if (!(*i)->isLibraryLoaded()) {
314  (*i)->loadLibrary();
315  }
316  if ((*i)->isClassAvailable<Base>(class_name)) {
317  return *i;
318  }
319  }
320  return nullptr;
321  }
