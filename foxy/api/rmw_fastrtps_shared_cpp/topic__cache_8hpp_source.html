205  bool removeTopic(
206  const eprosima::fastrtps::rtps::InstanceHandle_t & rtpsParticipantKey,
207  const eprosima::fastrtps::rtps::GUID_t & entity_guid,
208  const std::string & topic_name,
209  const std::string & type_name)
210  {
211  if (topic_name_to_topic_data_.find(topic_name) == topic_name_to_topic_data_.end()) {
212  RCUTILS_LOG_DEBUG_NAMED(
213  "rmw_fastrtps_shared_cpp",
214  "unexpected removal on topic '%s' with type '%s'",
215  topic_name.c_str(), type_name.c_str());
216  return false;
217  }
218  {
219  auto & type_vec = topic_name_to_topic_data_[topic_name];
220  type_vec.erase(
221  std::find_if(
222  type_vec.begin(), type_vec.end(),
223  [type_name, entity_guid](const auto & topic_data) {
224  return type_name.compare(topic_data.topic_type) == 0 &&
225  entity_guid == topic_data.entity_guid;
226  }));
227  if (type_vec.empty()) {
228  topic_name_to_topic_data_.erase(topic_name);
229  }
230  }
231  auto participant_guid = iHandle2GUID(rtpsParticipantKey);
232  auto guid_topics_pair = participant_to_topics_.find(participant_guid);
233  if (guid_topics_pair != participant_to_topics_.end() &&
234  guid_topics_pair->second.find(topic_name) != guid_topics_pair->second.end())
235  {
236  auto & type_vec = guid_topics_pair->second[topic_name];
237  type_vec.erase(std::find(type_vec.begin(), type_vec.end(), type_name));
238  if (type_vec.empty()) {
239  participant_to_topics_[participant_guid].erase(topic_name);
240  }
241  if (participant_to_topics_[participant_guid].empty()) {
242  participant_to_topics_.erase(participant_guid);
243  }
244  } else {
245  RCUTILS_LOG_ERROR_NAMED(
246  "rmw_fastrtps_shared_cpp",
247  "Unable to remove topic, does not exist '%s' with type '%s'",
248  topic_name.c_str(), type_name.c_str());
249  }
250  return true;
251  }
252 };
